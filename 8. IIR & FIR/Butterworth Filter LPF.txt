alphap=4;
alphas=30;
fs=400;
fp=800;
F=2000;
omp=2*fp/F;
oms=2*fs/F;
[n,wn]=buttord(omp,oms,alphap,alphas);
[b,a]=butter(n,wn);
w=0:0.01:pi;
[h,om]=freqz(b,a,w,'whole');
m=20*log(abs(h));
an=angle(h);
subplot(2,1,1);
plot(om/pi,m);
grid;
ylabel('Gain in dB');
xlabel('Normalised Frequency');
subplot(2,1,2);
plot(om/pi,an);
ylabel('Phase in radians');
xlabel('Normalised Frequency');
