x1=[1 2 3 4];
x2=[4 3 2 1];
xcorout=xcorr(x1,x2);
figure Name 'Cross Correlation';
subplot(1,3,1);
stem(x1);
title('1st Signal');
subplot(1,3,2);
stem(x2);
title('2nd Signal');
subplot(1,3,3);
stem(xcorout);
title('Cross Correlated Signal');
for i=1:3
subplot(1,3,i);
xlabel('Time');
ylabel('Amplitude');
end
